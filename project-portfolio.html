Code Sample

Example of the smooth scroll navigation script:

// Smooth scroll for navigation links const links = document.querySelectorAll('.nav-link[href^="#"]'); links.forEach(link => { link.addEventListener('click', (e) => { e.preventDefault(); const targetId = link.getAttribute('href'); const targetSection = document.querySelector(targetId); if (targetSection) { targetSection.scrollIntoView({ behavior: 'smooth' }); } }); });
